//check given string is palindrome or not using while-loop
let nam1="wowi";

function palindrome(nam1)
{
let left=0;
let right=nam1.length-1;
while(left<right){
if(nam1[left]!= nam1[right])
{
return false;
}
left++;
right--;
}return true;

}
console.log(palindrome(nam1));
